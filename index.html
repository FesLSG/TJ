    }

    form.addEventListener('submit', (e) => {

        e.preventDefault();

        const formData = new FormData(form);

        const newEntry = Object.fromEntries(formData.entries());


        newEntry.entry_price = newEntry.entry_price ? parseFloat(newEntry.entry_price) : null;

        newEntry.position_size = newEntry.position_size ? parseFloat(newEntry.position_size) : null;

        newEntry.stop_loss = newEntry.stop_loss ? parseFloat(newEntry.stop_loss) : null;

        newEntry.exit_price = newEntry.exit_price ? parseFloat(newEntry.exit_price) : null;

        newEntry.trade_id = tradeIdDisplay.value;


        const entries = loadEntries();

        const previousEquity = entries.length > 0 ? entries[0].equity : STARTING_EQUITY;


        if (newEntry.exit_price !== null) {

            const pL = newEntry.long_short === 'Long' ? (newEntry.exit_price - newEntry.entry_price) : (newEntry.entry_price - newEntry.exit_price);

            newEntry.pL = pL;


            if (pL > 0) {

                newEntry.result = 'Win';

            } else if (pL < 0) {

                newEntry.result = 'Loss';

            } else {

                newEntry.result = 'Break Even';

            }


            const risk = Math.abs(newEntry.entry_price - newEntry.stop_loss);

            if (risk > 0) {

                newEntry.rMultiple = pL / risk;

            } else {

                newEntry.rMultiple = null;

            }


            newEntry.equity = previousEquity + pL;

        } else {

            newEntry.pL = null;

            newEntry.result = null;

            newEntry.rMultiple = null;

            newEntry.equity = previousEquity;

        }


        entries.unshift(newEntry);

        saveEntries(entries);

        

        renderEntries();

        renderDashboard();

        renderHomeDashboard();

        form.reset();

        setCurrentDate();

        setCurrentTime();

        updateTradeIdDisplay();

    });

    function renderDashboard() {

        const allEntries = loadEntries().filter(e => e.result !== null);

        

        const assetFilter = dashboardAssetFilter.value;

        const positionFilter = dashboardPositionFilter.value;


        let filteredEntries = allEntries;

        if (assetFilter !== 'All') {

            filteredEntries = filteredEntries.filter(entry => entry.asset === assetFilter);

        }

        if (positionFilter !== 'All') {

            filteredEntries = filteredEntries.filter(entry => entry.long_short === positionFilter);

        }


        const totalPL = filteredEntries.reduce((sum, entry) => sum + entry.pL, 0);

        const wins = filteredEntries.filter(entry => entry.result === 'Win').length;

        const losses = filteredEntries.filter(entry => entry.result === 'Loss').length;

        const totalTrades = wins + losses;

        const winRate = totalTrades > 0 ? (wins / totalTrades) * 100 : 0;

        

        const rMultiples = filteredEntries.filter(entry => entry.rMultiple !== null).map(entry => entry.rMultiple);

        const avgRMultiple = rMultiples.length > 0 ? rMultiples.reduce((sum, r) => sum + r, 0) / rMultiples.length : 0;


        document.getElementById('total-pl-metric').textContent = `$${totalPL.toFixed(2)}`;

        document.getElementById('win-rate-metric').textContent = `${winRate.toFixed(2)}%`;

        document.getElementById('avg-r-multiple-metric').textContent = `${avgRMultiple.toFixed(2)}R`;


        const equityData = [STARTING_EQUITY];

        const labels = ['Starting Equity'];

        let currentEquity = STARTING_EQUITY;


        const entriesForChart = [...filteredEntries].reverse();

        entriesForChart.forEach((entry, index) => {

            currentEquity += entry.pL;

            equityData.push(currentEquity);

            labels.push(entry.trade_id);

        });


        const ctx = document.getElementById('equity-chart').getContext('2d');

        if (equityChart) {

            equityChart.destroy();

        }


        equityChart = new Chart(ctx, {

            type: 'line',

            data: {

                labels: labels,

                datasets: [{

                    label: 'Equity Curve',

                    data: equityData,

                    borderColor: 'rgb(52, 152, 219)',

                    backgroundColor: 'rgba(52, 152, 219, 0.2)',

                    fill: true,

                    tension: 0.1

                }]

            },

            options: {

                responsive: true,

                maintainAspectRatio: false,

                scales: {

                    x: {

                        display: true,

                        title: {

                            display: true,

                            text: 'Trades'

                        }

                    },

                    y: {

                        display: true,

                        title: {

                            display: true,

                            text: 'Equity ($)'

                        }

                    }

                }

            }

        });

    }


    function renderHomeDashboard() {

        const allEntries = loadEntries().filter(e => e.result !== null);

        const totalPL = allEntries.reduce((sum, entry) => sum + entry.pL, 0);

        const wins = allEntries.filter(entry => entry.result === 'Win');

        const losses = allEntries.filter(entry => entry.result === 'Loss');

        const totalClosedTrades = allEntries.length;


        const totalGrossProfit = wins.reduce((sum, entry) => sum + entry.pL, 0);

        const totalGrossLoss = losses.reduce((sum, entry) => sum + entry.pL, 0);


        const profitFactor = totalGrossLoss !== 0 ? Math.abs(totalGrossProfit / totalGrossLoss) : (totalGrossProfit > 0 ? Infinity : 0);

        const strikeRate = totalClosedTrades > 0 ? (wins.length / totalClosedTrades) * 100 : 0;

        const finalEquity = allEntries.length > 0 ? allEntries[0].equity : STARTING_EQUITY;


        // Update metric cards

        document.getElementById('net-pnl').textContent = `$${totalPL.toFixed(2)}`;

        document.getElementById('profit-factor-metric').textContent = profitFactor !== Infinity ? profitFactor.toFixed(2) : 'N/A';

        document.getElementById('strike-rate-metric').textContent = `${strikeRate.toFixed(2)}%`;

        document.getElementById('account-balance').textContent = `$${finalEquity.toFixed(2)}`;


        // Render profitability calendar

        renderProfitCalendar();

    }

    

    function renderProfitCalendar() {

        const entries = loadEntries().filter(e => e.result !== null);

        const calendarEl = document.getElementById('profit-calendar');

        const monthYearEl = document.getElementById('current-month-year');

        calendarEl.innerHTML = ''; // Clear previous calendar


        const year = currentCalendarDate.getFullYear();

        const month = currentCalendarDate.getMonth();

        

        monthYearEl.textContent = currentCalendarDate.toLocaleString('default', { month: 'long', year: 'numeric' });

        

        const firstDayOfMonth = new Date(year, month, 1).getDay(); // 0 = Sunday, 1 = Monday, etc.

        const daysInMonth = new Date(year, month + 1, 0).getDate();


        const pnlByDay = {};

        entries.forEach(entry => {

            const entryDate = new Date(entry.date);

            if (entryDate.getFullYear() === year && entryDate.getMonth() === month) {

                if (!pnlByDay[entry.date]) {

                    pnlByDay[entry.date] = 0;

                }

                pnlByDay[entry.date] += entry.pL;

            }

        });


        const dayNames = ['Sun', 'Mon', 'Tue', 'Wed', 'Thu', 'Fri', 'Sat'];

        dayNames.forEach(dayName => {

            const headerCell = document.createElement('div');

            headerCell.className = 'calendar-day calendar-header';

            headerCell.textContent = dayName;

            calendarEl.appendChild(headerCell);

        });


        for (let i = 0; i < firstDayOfMonth; i++) {

            const blankCell = document.createElement('div');

            blankCell.className = 'calendar-day';

            calendarEl.appendChild(blankCell);

        }


        for (let day = 1; day <= daysInMonth; day++) {

            const dateStr = `${year}-${String(month + 1).padStart(2, '0')}-${String(day).padStart(2, '0')}`;

            const dayCell = document.createElement('div');

            dayCell.className = 'calendar-day';


            const dayNum = document.createElement('span');

            dayNum.className = 'day-number';

            dayNum.textContent = day;

            dayCell.appendChild(dayNum);


            if (pnlByDay[dateStr] !== undefined) {

                const pnl = pnlByDay[dateStr];

                const pnlSpan = document.createElement('span');

                pnlSpan.className = 'day-pnl';

                if (pnl > 0) {

                    pnlSpan.classList.add('positive-pnl');

                    pnlSpan.textContent = `$${pnl.toFixed(2)}`;

                } else if (pnl < 0) {

                    pnlSpan.classList.add('negative-pnl');

                    pnlSpan.textContent = `-$${Math.abs(pnl).toFixed(2)}`;

                } else {

                    pnlSpan.textContent = `$${pnl.toFixed(2)}`;

                }

                dayCell.appendChild(pnlSpan);

            }

            calendarEl.appendChild(dayCell);

        }

    }
